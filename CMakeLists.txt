cmake_minimum_required(VERSION 3.23)

project(ocmake_util)
list(APPEND CMAKE_MODULE_PATH "${CMAKE_CURRENT_SOURCE_DIR}" "${CMAKE_CURRENT_SOURCE_DIR}/Modules" "${CMAKE_CURRENT_SOURCE_DIR}/Projects" "${CMAKE_CURRENT_SOURCE_DIR}/Program")
list(REMOVE_DUPLICATES CMAKE_MODULE_PATH)
set(CMAKE_MODULE_PATH ${CMAKE_MODULE_PATH} CACHE INTERNAL "CMAKE_MODULE_PATH")
set(OCMAKEUTIL_PATH ${CMAKE_CURRENT_SOURCE_DIR} CACHE INTERNAL "OCMAKEUTIL_PATH")
if(IMPORT_PROJECT_EXTERNAL_DIR)
  cmake_path(NORMAL_PATH IMPORT_PROJECT_EXTERNAL_DIR)
  set(IMPORT_PROJECT_EXTERNAL_DIR ${IMPORT_PROJECT_EXTERNAL_DIR} CACHE INTERNAL "IMPORT_PROJECT_EXTERNAL_DIR" FORCE)
endif()
set(OCMAKEUTIL_PROJECTS_PATH "${CMAKE_CURRENT_SOURCE_DIR}/Projects" CACHE INTERNAL "OCMAKEUTIL_PROJECTS_PATH")
include(ProjectDefinition)
